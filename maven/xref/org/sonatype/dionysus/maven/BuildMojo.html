13  package org.sonatype.dionysus.maven;
14  
15  import java.io.File;
16  import java.util.ArrayList;
17  import java.util.List;
18  
19  import org.apache.maven.plugins.annotations.Mojo;
20  import org.apache.maven.plugins.annotations.Parameter;
21  import org.apache.maven.project.MavenProject;
22  
23  /**
24   * Build site content.
25   *
26   * @since 1.0.0
27   */
28  @Mojo(name="build", requiresDirectInvocation = true, aggregator = true)
29  class BuildMojo
30    extends MojoSupport
31  {
32    @Parameter(defaultValue = "${reactorProjects}", required = true, readonly = true)
33    private List<MavenProject> projects;
34  
35    /**
36     * Coordinates to theme bundle.
37     *
38     * This needs to be {@code zip} artifact.
39     */
40    @Parameter
41    ThemeCoordinate theme;
42  
43    /**
44     * Name of theme.
45     *
46     * When used with {@link #theme} parameter this will be the directory the theme bundle will be installed.
47     */
48    @Parameter(property = "dionysus.themeName", defaultValue = "dionysus", required = true)
49    String themeName;
50  
51    /**
52     * Rebuild the theme.  This requires additional setup to function.
53     */
54    @Parameter(property = "dionysus.rebuildTheme", defaultValue = "false")
55    boolean rebuildTheme;
56  
57    /**
58     * Generate Hugo content.
59     */
60    @Parameter(property = "dionysus.generateHugo", defaultValue = "true")
61    boolean generateHugo;
62  
63    /**
64     * Hugo executable.
65     *
66     * This can be overridden to a locally installed Hugo, or by default will extract a binary for the platform.
67     *
68     * Only supports Mac and Linux.
69     */
70    @Parameter(property = "dionysus.hugoExecutable", defaultValue = "${project.build.directory}/hugo", required = true)
71    File hugoExecutable;
72  
73    /**
74     * Additional options to pass to the {@code hugo} executable.
75     */
76    @Parameter
77    List<String> hugoOptions = new ArrayList<>();
78  
79    /**
80     * Location of the Hugo site.
81     */
82    @Parameter(property = "dionysus.hugoDir", defaultValue = "${project.basedir}/src/site/hugo", required = true)
83    File hugoDir;
84  
85    /**
86     * Generate Maven content.
87     *
88     * Maven execution will have {@code dionysus} {@code dionysus-generate-maven-site} profiles enabled
89     * and include any configured {@link #mavenOptions}.
90     */
91    @Parameter(property = "dionysus.generateMaven", defaultValue = "true")
92    boolean generateMaven;
93  
94    /**
95     * Additional options to pass to Maven executions.
96     *
97     * @see #generateMaven
98     * @see #generateData
99     */
100   @Parameter
101   List<String> mavenOptions = new ArrayList<>();
102 
103   /**
104    * The goals which will be executed to generate Maven content.
105    */
106   @Parameter
107   List<String> mavenGoals = new ArrayList<>();
108 
109   /**
110    * Generate Hugo data with {@code org.sonatype.goodies.dionysus:dionysus-maven-plugin:generate-data}.
111    *
112    * Maven execution will have {@code dionysus} and {@code dionysus-generate-data} profiles enabled
113    * and include any configured {@link #mavenOptions}.
114    */
115   @Parameter(property = "dionysus.generateData", defaultValue = "true")
116   boolean generateData;
117 
118   /**
119    * Collect generated Hugo data.
120    */
121   @Parameter(property = "dionysus.collectData", defaultValue = "true")
122   boolean collectData;
123 
124   /**
125    * Where generate Hugo data will be collected to.
126    */
127   @Parameter(property = "dionysus.hugoData", defaultValue = "${project.basedir}/src/site/hugo/data", required = true)
128   File hugoData;
129 
130   @Override
131   protected void doExecute() {
132     new BuildTask(this).execute();
133   }
134 }
